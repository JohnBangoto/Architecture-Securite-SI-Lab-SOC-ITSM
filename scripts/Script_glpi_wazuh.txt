#!/usr/bin/env python3
# custom-glpi - Integration Wazuh -> GLPI

import sys
import json
import requests
import urllib3

urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning)

# ─── CONFIGURATION ────────────────────────────────────────
GLPI_URL    = "http://185.229.225.xxx/apirest.php"
APP_TOKEN   = "VOTRE_APP_TOKEN"
USER_TOKEN  = "VOTRE_USER_TOKEN"
# ──────────────────────────────────────────────────────────

def get_session_token():
    headers = {
        "Content-Type": "application/json",
        "Authorization": f"user_token {USER_TOKEN}",
        "App-Token": APP_TOKEN
    }
    response = requests.get(
        f"{GLPI_URL}/initSession",
        headers=headers,
        verify=False
    )
    return response.json().get("session_token")

def create_ticket(session_token, title, content, priority):
    headers = {
        "Content-Type": "application/json",
        "Session-Token": session_token,
        "App-Token": APP_TOKEN
    }
    # Mapping priorité Wazuh → GLPI
    # Wazuh: 1-15 | GLPI: 1(très bas) à 6(très haute)
    if priority >= 13:
        glpi_priority = 6    # Très haute
    elif priority >= 10:
        glpi_priority = 5    # Haute
    elif priority >= 7:
        glpi_priority = 4    # Moyenne haute
    else:
        glpi_priority = 3    # Moyenne

    payload = {
        "input": {
            "name": title,
            "content": content,
            "priority": glpi_priority,
            "itilcategories_id": 0,   # 0 = aucune catégorie
            "type": 1                  # 1=Incident, 2=Demande
        }
    }
    response = requests.post(
        f"{GLPI_URL}/Ticket",
        headers=headers,
        json=payload,
        verify=False
    )
    return response.json()

def kill_session(session_token):
    headers = {
        "Session-Token": session_token,
        "App-Token": APP_TOKEN
    }
    requests.get(
        f"{GLPI_URL}/killSession",
        headers=headers,
        verify=False
    )

def main():
    # Wazuh passe le fichier alerte en argument
    alert_file = sys.argv[1]

    with open(alert_file) as f:
        alert = json.load(f)

    # Extraction des infos
    level       = alert.get("rule", {}).get("level", 0)
    description = alert.get("rule", {}).get("description", "Alerte Wazuh")
    agent_name  = alert.get("agent", {}).get("name", "Unknown")
    agent_ip    = alert.get("agent", {}).get("ip", "Unknown")
    rule_id     = alert.get("rule", {}).get("id", "Unknown")
    timestamp   = alert.get("timestamp", "Unknown")

    title = f"[Wazuh] {description} - Agent: {agent_name}"
    content = f"""
Alerte Wazuh détectée :

- Agent     : {agent_name} ({agent_ip})
- Règle ID  : {rule_id}
- Niveau    : {level}
- Date      : {timestamp}
- Détail    : {description}

Données complètes :
{json.dumps(alert, indent=2, ensure_ascii=False)}
    """.strip()

    # Créer le ticket
    session = get_session_token()
    if not session:
        sys.exit(1)

    result = create_ticket(session, title, content, level)
    kill_session(session)

    print(f"Ticket créé : {result}")

if __name__ == "__main__":
    main()
